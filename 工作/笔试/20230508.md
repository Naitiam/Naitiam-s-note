```
  function Person(name, age) {
    this.name = name;
    this.age = age;
    this.eat = function () {};
  }
  Person.run = function () {};
  Person.prototype.walk = function () {};
  const p1 = new Person("Tom", 22);
  const p2 = new Person("Jhon", 23);
  console.log(p1.eat === p2.eat);
  console.log(p1.run === p2.run);
  console.log(p1.walk === p2.walk);
```
