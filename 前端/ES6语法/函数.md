## 默认值

ES6允许为函数的参数设置默认值，即直接写在参数定义的后面。

```javascript
    // function add(a , b){
    //     //es2020写法
    //     a ||= 10;
    //     b ||= 10;
    //     return a + b;
    // }
    //默认值
    // function add(a = 20,b = 10){
    //     return a + b;
    // }
    //默认的表达式值可以为一个函数
    //这里去看看官方说法
    function add(a, b = getVal(5)){
        return a + b;
    }
    function getVal(val){
        return val + 5;
    }
    console.log(add(10));
```

## 剩余运算符

以往使用arguments函数来处理传进来的多个参数。

**arguments**对象存储的是**实际传递给函数的参数**。是个类似数组但不是数组的对象，说他类似数组是因为其具备数组相同的访问性质及方式，**能够由arguments[n]来访问对应的单个参数的值**，并**拥有数组长度属性length**。

```javascript
    let book = {
        title: 'es6语法',
        year: 2020
    }
    let bookData = pick(book,'year','title')
    function pick(obj){
        console.log(arguments);
        let result = Object.create(null);
        for(let i = 1;i < arguments.length;i++){
            result[arguments[i]] = obj[arguments[i]]
        }
        return result;
    }
```

**剩余参数**：三个点  `...参数名`  例如 `...keys`

```javascript
    function pick(obj, ...keys ){
        let result = Object.create(null);
        for(let i = 0;i < keys.length;i++){
            console.log(keys[i]);
            console.log(obj[keys[i]]);
            result[keys[i]] = obj[keys[i]]
        }
        return result;
    }
    console.log(bookData);
```

## 扩展运算符

> 将一个数组分割，并将各项作为分离的参数传递给函数‘

```javascript
    const arr = [54,98,100,54,62,37]
    //遍历取最大值
    //处理数组最大值，使用apply
    //apply???是啥
    console.log(Math.max.apply(null,arr));
    //扩展运算符
    console.log(Math.max(...arr));
```

## 箭头函数

> 快捷键 an+TAB

- this 是静态的，this始终指向函数声明时所在作用域下的this的值
- 不能作为构造实例化对象
- 不能使用 arguments 变量保存实参

```javascript
    // let add = function(x, y){
    //     return x + y;
    // }
    // let add = (x,y)=>{
    //     return x + y;
    // }

    //箭头右边默认return，若是对象需要()
    let val = val => val + 5;
    console.log(val(5));

    let obj = id => ({
        id: id,
        name: '张三'
    })
    console.log(obj(1));

    //函数需要return返回
    let fn = (() => {
        return () => {
            console.log("套")
        }
    })();
    fn();

    //箭头函数this的指向
    let PageHandle = {
        name: '测试',
        init: function () {
            // document.addEventListener('click',function(event){
            //     //这里this指向document对象
            //     console.log(this);
            //     this.doSomeTings(event.type);
            // }.bind(this),false)
            //bind es5 写法，更改this指向 到父级
            document.addEventListener('click', (event) => {
                this.doSomeTings(event.type)
            }, false)
        },
        // doSomeTings: function(type){
        //     console.log(this);
        //     console.log(`事件类型${type},name:${this.name}`);
        // }
        doSomeTings: (type) => {
            //此处this指向windows对象
            console.log(this);
            console.log(`事件类型${type},name:${this.name}`)
        }
    }
    PageHandle.init();
    //this指向外层(父级)作用域，若没有指向windows对象
    //在vue中，大多要用箭头函数指向vue实例对象
    //this是静态的无法构造
    //内部无法使用arguments

    //找数组中的偶数
    let item = [24, 53, 75, 87, 98];
    // let res = item.filter(function(item){
    //     return item % 2 === 0;
    // });
    let res = item.filter(item => item % 2 === 0);
    console.log(res);

    //点击改变颜色
    let ad = document.getElementById('box');
    ad.addEventListener("click",function(){
        setTimeout(()=>{  
            this.style.background = 'aquamarine';
        },2000)
    },false)
```
